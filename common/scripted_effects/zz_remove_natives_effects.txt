remove_migratory_natives_NA = {
    every_province = {
        limit = {
            superregion = north_america
            owner = { is_primitive = yes }   # robust across patches
        }
        decolonize = yes
    }
}

# Use this if you also want South America, but keep the Andes intact
remove_migratory_natives_SA_excl_Andes = {
    every_province = {
        limit = {
            superregion = south_america
            NOT = { region = andes }
            owner = { is_primitive = yes }
        }
        decolonize = yes
    }
}
